CREATE VIEW v_inventory_status AS
SELECT 
    i.inventory_id,
    i.product_id,
    CAST(p.category AS VARCHAR(100)) AS category,
    CAST(p.brand AS VARCHAR(100)) AS brand,
    i.warehouse_id,
    i.opening_stock,
    i.closing_stock,

    -- Stock usage
    (i.opening_stock - i.closing_stock) AS stock_usage,

    -- Average inventory
    ((i.opening_stock + i.closing_stock) / 2.0) AS average_inventory,

    -- Movement category (Fast / Medium / Slow / Zero)
    CASE 
        WHEN COALESCE(SUM(s.quantity_sold),0) > 1000 THEN 'Fast Moving'
        WHEN COALESCE(SUM(s.quantity_sold),0) BETWEEN 500 AND 1000 THEN 'Medium Moving'
        WHEN COALESCE(SUM(s.quantity_sold),0) BETWEEN 1 AND 499 THEN 'Slow Moving'
        ELSE 'Zero Movement'
    END AS movement_category,

    -- Stock status (Adjusted thresholds)
    CASE 
        WHEN i.closing_stock > 1500 THEN 'Overstock'
        WHEN i.closing_stock < 500 THEN 'Understock'
        ELSE 'Optimal'
    END AS stock_status

FROM inventory i
JOIN products p 
    ON i.product_id = p.product_id
LEFT JOIN sales s 
    ON i.product_id = s.product_id
GROUP BY 
    i.inventory_id,
    i.product_id,
    CAST(p.category AS VARCHAR(100)),
    CAST(p.brand AS VARCHAR(100)),
    i.warehouse_id,
    i.opening_stock,
    i.closing_stock;
GO

CREATE VIEW v_supplier_performance AS
SELECT 
    s.supplier_id,
    CAST(s.supplier_name AS VARCHAR(200)) AS supplier_name,
    s.avg_lead_time_days,

    CASE 
        WHEN s.avg_lead_time_days < 8 THEN 'Rapid'
        WHEN s.avg_lead_time_days BETWEEN 8 AND 15 THEN 'Moderate'
        ELSE 'Late'
    END AS supplier_category,

    COUNT(po.po_id) AS total_orders
FROM suppliers s
LEFT JOIN purchase_orders po 
    ON s.supplier_id = po.supplier_id
GROUP BY 
    s.supplier_id,
    CAST(s.supplier_name AS VARCHAR(200)),
    s.avg_lead_time_days;

    CAST(p.category AS VARCHAR(200)) AS category,
    SUM(s.quantity_sold) AS total_sales_units,   -- ✅ Units sold, not revenue

    -- Return rate (% of units returned)
    (COUNT(sr.return_id) * 100.0 / NULLIF(COUNT(s.sale_id),0)) AS return_rate_percent,

    sr.return_reason,
    COUNT(sr.return_id) AS total_returns

FROM sales s
JOIN products p 
    ON s.product_id = p.product_id
LEFT JOIN sales_returns sr 
    ON s.sale_id = sr.sale_id
GROUP BY 
    CAST(p.category AS VARCHAR(200)),
    sr.return_reason;

    CAST(p.brand AS VARCHAR(100)) AS brand,
    c.customer_type,

    SUM(s.revenue) AS total_revenue

    ON s.customer_id = c.customer_id

    p.product_id, 
    CAST(p.category AS VARCHAR(100)), 
    CAST(p.brand AS VARCHAR(100)),
    c.customer_type,
    c.region,
